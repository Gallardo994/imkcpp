50  } else {

52  const i32 delta = std::abs(rtt - static_cast<i32>(this->srtt));
53 
54  // RTTVAR = (1 - BETA) * RTTVAR + BETA * |SRTT - RTT|
55  // Simplified to: RTTVAR = (BETA_MUL * RTTVAR + delta) / BETA_DIV
56  // In RFC 2988, BETA = 1/4
57  constexpr u32 BETA_MUL = 3;
58  constexpr u32 BETA_DIV = 4;
59  this->rttvar = (BETA_MUL * this->rttvar + delta) / BETA_DIV;
60 
61  // SRTT = (1 - ALPHA) * SRTT + ALPHA * RTT
62  // Simplified to: SRTT = (ALPHA_MUL * SRTT + RTT) / ALPHA_DIV
63  // In RFC 2988, ALPHA = 1/8
64  constexpr u32 ALPHA_MUL = 7;
65  constexpr u32 ALPHA_DIV = 8;
66  this->srtt = (ALPHA_MUL * this->srtt + rtt) / ALPHA_DIV;
67  }
68 
69  // RTO = SRTT + max(G, K * RTTVAR)
70  constexpr u32 K = 4;
71  const u32 rto = this->srtt + std::max(this->interval, K * this->rttvar);
72 
73  // Limiting to IKCP_RTO_MAX is not mentioned in RFC 2988, but is present in the original C implementation
74  this->rto = std::clamp(rto, this->minrto, this->maxrto);
