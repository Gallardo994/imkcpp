52  [[nodiscard]] tl::expected<size_t, error> send(const std::span<const std::byte> buffer) {
53  if (buffer.empty()) {
54  return tl::unexpected(error::buffer_too_small);
55  }
56 
57  const size_t count = estimate_segments_count(buffer.size());
58 
59  if (count > std::numeric_limits<u8>::max()) {
60  return tl::unexpected(error::too_many_fragments);
61  }
62 
63  // We should limit sending by receive window because we can send more than send_wnd segments,
64  // but cannot receive more than rcv_wnd segments.
65  if (count > this->congestion_controller.get_receive_window()) {
66  return tl::unexpected(error::exceeds_window_size);
67  }
68 
69  size_t offset = 0;
70 
71  for (size_t i = 0; i < count; i++) {
72  const size_t size = std::min(buffer.size() - offset, MAX_SEGMENT_SIZE);
73  assert(size > 0);
74 
75  Segment& seg = this->snd_queue.emplace_back();
76  seg.data_assign({buffer.data() + offset, size});
77  seg.header.frg = static_cast<u8>(count - i - 1);
78 
79  assert(seg.data_size() == size);
80 
81  offset += size;
82  }
83 
84  return offset;
85  }

132  void flush_data_segments(FlushResult& flush_result, const output_callback_t& output, const u32 current, const i32 unused_receive_window) {
133  const u32 cwnd = this->congestion_controller.calculate_congestion_window();
134  this->move_send_queue_to_buffer(cwnd, current, unused_receive_window);
135 
136  bool change = false;
137  bool lost = false;
138 
139  const u32 resent = (this->fastresend > 0) ? this->fastresend : 0xffffffff;
140  const u32 rtomin = (this->nodelay == 0) ? (this->rto_calculator.get_rto() >> 3) : 0;
141 
142  auto& snd_buf = this->sender_buffer.get();
143 
144  for (Segment& segment : snd_buf) {
145  bool needsend = false;
146 
147  if (segment.metadata.xmit == 0) {
148  needsend = true;
149  segment.metadata.xmit++;
150  segment.metadata.rto = this->rto_calculator.get_rto();
151  segment.metadata.resendts = current + segment.metadata.rto + rtomin;
152  } else if (time_delta(current, segment.metadata.resendts) >= 0) {
153  needsend = true;
154  segment.metadata.xmit++;
155  ++this->xmit;
156 
157  if (this->nodelay == 0) {
158  segment.metadata.rto += std::max(segment.metadata.rto, this->rto_calculator.get_rto());
159  } else {
160  const u32 step = this->nodelay < 2? segment.metadata.rto : this->rto_calculator.get_rto();
161  segment.metadata.rto += step / 2;
162  }
163 
164  segment.metadata.resendts = current + segment.metadata.rto;
165  lost = true;
166 
167  flush_result.timeout_retransmitted_count++;
168  } else if (resent < segment.metadata.fastack) {
169  if (segment.metadata.xmit <= this->fastlimit || this->fastlimit == 0) {
170  needsend = true;
171  change = true;
172 
173  segment.metadata.xmit++;
174  segment.metadata.fastack = 0;
175  segment.metadata.resendts = current + segment.metadata.rto;
176 
177  flush_result.fast_retransmitted_count++;
178  }
179  }
180 
181  if (needsend) {
182  segment.header.ts = current;
183  segment.header.wnd = unused_receive_window;
184  segment.header.una = this->segment_tracker.get_rcv_nxt();
185 
186  flush_result.total_bytes_sent += this->flusher.flush_if_does_not_fit(output, segment.data_size());
187  this->flusher.emplace_segment(segment);
188 
189  if (segment.metadata.xmit >= this->dead_link) {
190  this->shared_ctx.set_state(State::DeadLink);
191  }
192 
193  flush_result.cmd_push_count++;
194  }
195  }
196 
197  if (change) {
198  this->congestion_controller.packets_resent(this->segment_tracker.get_packets_in_flight_count(), resent);
199  }
200 
201  if (lost) {
202  this->congestion_controller.packet_lost();
203  }
204  }
205  };
