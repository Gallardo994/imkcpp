121  auto input(const std::span<const std::byte> data) noexcept -> tl::expected<InputResult, error> {
122  if (data.size() < constants::IKCP_OVERHEAD) {
123  return tl::unexpected(error::less_than_header_size);
124  }

127 
128  const u32 prev_una = this->segment_tracker.get_snd_una();
129  FastAckCtx fastack_ctx{};
130 
131  SegmentHeader header;
132  size_t offset = 0;
133 
134  const auto drop_push = [&] {
135  offset += header.len;
136  input_result.dropped_push_count++;
137  };
138 
139  const auto data_size = data.size();

141  while (true) {

143  break;
144  }
145 
146  header.decode_from(data, offset);
147 
148  if (header.conv != this->shared_ctx.get_conv()) {
149  return tl::unexpected(error::conv_mismatch);
150  }
151 
152  if (header.len > data.size() - offset) {
153  return tl::unexpected(error::header_and_payload_length_mismatch);
154  }
155 
156  if (!commands::is_valid(header.cmd)) {
157  return tl::unexpected(error::unknown_command);
158  }
159 
160  this->congestion_controller.set_remote_window(header.wnd);
161  this->ack_controller.una_received(header.una);
162 
163  switch (header.cmd) {
164  case commands::IKCP_CMD_PUSH: {
165  if (!this->congestion_controller.fits_receive_window(header.sn)) {
166  drop_push();
167  break;
168  }
169 
170  this->ack_controller.schedule_ack(header.sn, header.ts);
171 
172  if (this->segment_tracker.should_receive(header.sn)) {
173  SegmentData segment_data;
174  segment_data.decode_from(data, offset, header.len);
175 
176  this->receiver_buffer.emplace_segment(header, segment_data);
177  this->receiver.move_receive_buffer_to_queue();
178  } else {
179  drop_push();
180  }
181  break;
182  }
183  case commands::IKCP_CMD_ACK: {
184  this->ack_controller.ack_received(this->current, header.sn, header.ts);
185  fastack_ctx.update(header.sn, header.ts);
186  input_result.cmd_ack_count++;
187  break;
188  }
189  case commands::IKCP_CMD_WASK: {
190  this->congestion_controller.set_probe_flag(constants::IKCP_ASK_TELL);
191  input_result.cmd_wask_count++;
192  break;
193  }
194  case commands::IKCP_CMD_WINS: {
195  input_result.cmd_wins_count++;
196  break;
197  }
198  default: {
199  return tl::unexpected(error::unknown_command);
200  }
201  }
202  }
203

205 
206  if (this->segment_tracker.get_snd_una() > prev_una) {
207  this->congestion_controller.adjust_parameters();
208  }

211

213  }

250  }

287  FlushResult flush_result{};
288 
289  if (!this->updated) {
290  return flush_result;

300 
301  // Window probes
302  this->congestion_controller.update_probe_request(current);

304

307 
308  // Flush remaining
309  flush_result.total_bytes_sent += this->flusher.flush_if_not_empty(callback);
310 
311  this->congestion_controller.ensure_at_least_one_packet_in_flight();
312
